## 2020 春季 HUNNU第二次训练

**A Rational Sequence (Take 3)**

题意：一棵树，在1号根节点，p=1,q=1,其值为 p/q。每个节点的左节点为 p/（p+q） ，右节点为 （p+q）/q。要你求n号节点的值。

分析：很简单，用栈保存，然后一步步往上存就可以了。

```c++
#include <bits/stdc++.h>
using namespace std;
int n, p = 1, q = 1, id, cnt;
stack<int> Q;
int main()
{
    scanf("%d", &n);
    while (n--)
    {
        p = q = 1;
        scanf(" %d %d", &cnt, &id);
        while (id > 1)
        {
            if (id % 2)
                Q.push(1); //1右
            else
                Q.push(0); //0左
            id /= 2;
        }
        while (!Q.empty())
        {
            int x = Q.top();
            Q.pop();
            if (x)
                p = p + q;
            else
                q = p + q;
        }
        cout << cnt << ' ' << p << '/' << q << endl;
    }
    return 0;
}
```
