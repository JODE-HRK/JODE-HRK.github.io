# POJ 1961 Period

题意： 给你无数个字符串，要你求每个字符串的每个前缀是否都能有一个串重复而成

分析：求最小循环节，并且当循环节的长度能被前缀的长度整除的时候输出

```c++
#include <bits/stdc++.h>
using namespace std;
const int maxn = 1e6 + 7;
int nxt[maxn], n;
void getnxt(string s)
{
    nxt[0] = -1;
    int i = 0, j = -1;
    while (i < n)
    {
        if (j == -1 || s[i] == s[j])
            nxt[++i] = ++j;
        else
            j = nxt[j];
    }
}
int main()
{
    int cnt = 0;
    string s;
    while (cin >> n)
    {
        if (n == 0)
            break;
        cin >> s;
        cout << "Test case #" << ++cnt << endl;
        getnxt(s);
        for (int i = 1; i <= n; i++)
            if (nxt[i] > 0 && i % (i - nxt[i]) == 0)
                cout << i << ' ' << i / (i - nxt[i]) << endl;
        cout << endl;
    }
    return 0;
}
```
